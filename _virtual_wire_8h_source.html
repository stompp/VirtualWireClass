25 // Maximum number of bytes in a message, counting the byte count and FCS
26 #define VW_MAX_MESSAGE_LEN 30
27 
28 // The maximum payload length
29 #define VW_MAX_PAYLOAD VW_MAX_MESSAGE_LEN-3
30 
31 // The size of the receiver ramp. Ramp wraps modulu this number
32 #define VW_RX_RAMP_LEN 160
33 
34 // Number of samples per bit
35 #define VW_RX_SAMPLES_PER_BIT 8
36 
37 // Ramp adjustment parameters
38 // Standard is if a transition occurs before VW_RAMP_TRANSITION (80) in the ramp,
39 // the ramp is retarded by adding VW_RAMP_INC_RETARD (11)
40 // else by adding VW_RAMP_INC_ADVANCE (29)
41 // If there is no transition it is adjusted by VW_RAMP_INC (20)
42 #define VW_RAMP_INC (VW_RX_RAMP_LEN/VW_RX_SAMPLES_PER_BIT)
43 #define VW_RAMP_TRANSITION VW_RX_RAMP_LEN/2
44 #define VW_RAMP_ADJUST 9
45 #define VW_RAMP_INC_RETARD (VW_RAMP_INC-VW_RAMP_ADJUST)
46 #define VW_RAMP_INC_ADVANCE (VW_RAMP_INC+VW_RAMP_ADJUST)
47 
48 // Outgoing message bits grouped as 6-bit words
49 // 36 alternating 1/0 bits, followed by 12 bits of start symbol
50 // Followed immediately by the 4-6 bit encoded byte count,
51 // message buffer and 2 byte FCS
52 // Each byte from the byte count on is translated into 2x6-bit words
53 // Caution, each symbol is transmitted LSBit first,
54 // but each byte is transmitted high nybble first
55 #define VW_HEADER_LEN 8
